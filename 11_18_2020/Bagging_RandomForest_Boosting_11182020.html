set.seed(1)
shrink <- seq(0.0001, 0.10, length.out = 10)
train.mse <- double(10)
for (i in 1:10) {
  boost.hitters <-
    gbm(
      Salary ~ .,
      data = dat.train,
      distribution = "gaussian",
      n.trees = 1000,
      shrinkage = shrink[i],
      interaction.depth = 4
    )
  yhat.boost <-
    predict(boost.hitters, newdata = dat.train, n.trees = 1000)
  train.mse[i] <- mean((yhat.boost - dat.train$Salary) ^ 2)
}
